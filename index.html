const express = require('express'); const http = require('http'); const { Server } = require('socket.io'); const app = express(); const server = http.createServer(app); const io = new Server(server); // The HTML, CSS, and Client-side JavaScript combined into one single string const htmlTemplate = ` All-in-One Chat Box

💬 Friends Chat Room

    `; // Serve the mixed HTML string directly when someone visits the site app.get('/', (req, res) => { res.send(htmlTemplate); }); // Manage user connections and real-time messages io.on('connection', (socket) => { console.log('User connected'); socket.on('chat message', (msg) => { io.emit('chat message', msg); // Broadcasts message to everyone }); socket.on('disconnect', () => { console.log('User disconnected'); }); }); // Start the server const PORT = 3000; server.listen(PORT, () => { console.log(`Server running! Go to http://localhost:${PORT}`); });
